#!/bin/bash

echo "DevOps Practice (Short Script)"

# System Info
echo "User: $(whoami)"
echo "Current Dir: $(pwd)"

# Create Project
mkdir demo_project
cd demo_project

# File Creation
echo "Hello DevOps" > app.txt

# File Operations
cp app.txt backup.txt

# Permissions
chmod 744 app.txt

# Git Practice
git init
git add app.txt
git commit -m "Initial commit"

# Show Logs
git log --oneline

echo "Completed"
